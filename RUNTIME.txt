def update
  @step = params[:step] || 'agency_info'
  if @agency.update(agency_params)
    case @step
    when 'agency_info' then redirect_to agency_onboarding_path(@agency, step: 'system_setup')
    when 'system_setup' then redirect_to agency_onboarding_path(@agency, step: 'users')
    when 'users'
      @agency.update!(status: 'active')
      redirect_to agency_path(@agency), notice: 'Agency successfully onboarded.'
    end
  else
    render :show, status: :unprocessable_entity
  end
end
